      nodes {
        id
        mediaContentType
        alt

          }

      }

    variants(first: 100) {
      nodes {
        id
        title
        price
        compareAtPrice
        sku
        barcode
        inventoryPolicy

          id        # ← ADD THIS LINE

            }

        }
      }
    }
  }
}
